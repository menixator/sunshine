let humanize = value => {
  return value.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
};

exports.power = value => {
  if (value > 1000000) {
    return {
      value,
      humanizedUnit: "MWh",
      humanized: humanize(fix(value / 1000000))
    };
  }

  if (value > 1000) {
    return {
      value,
      humanizedUnit: "KWh",
      humanized: humanize(fix(value / 1000))
    };
  }

  return { value, humanized: humanize(fix(value)), humanizedUnit: "Wh" };
};
